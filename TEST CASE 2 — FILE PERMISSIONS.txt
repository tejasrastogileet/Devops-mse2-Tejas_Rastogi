# Step 1: Go to home directory
cd ~

# Step 2: Create a new file
touch index.html

# Step 3: Add content to the file
echo "<h1>Hello LifeBridge</h1>" > index.html

# Step 4: Check current file permissions
ls -l index.html

# Step 5: Give full permissions (read, write, execute to all)
chmod 777 index.html

# Step 6: Verify updated permissions
ls -l index.html

# Step 7: Change file owner to current user
chown $USER:$USER index.html

# Step 8: Set secure permissions (read/write for owner, read/execute for others)
chmod 755 index.html

# Step 9: Final verification of permissions
ls -l index.html
